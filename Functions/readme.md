function Hello(parameter_1, para_2,para_3 ..) {
   let return_value;
  // Function code
  return return_value;

```

## Anonymous Functions

```javascript
// Assignment of an anonymous function to the myFunc variable
const myFunc = (param1, param2, ...) => {
  // Statements using param1, param2, ...
};


// Anonymous function call
// param1 value is set to arg1, param2 to arg2, ...
myFunc(arg1, arg2, ...);




// These can be further simplified using arrow functions
// For Ex:

let sum = (var1,var2) => {
   
   let ans=var1+var2;
   return ans;

}

console.log(sum(2,3));

// Output:
// 5

``
